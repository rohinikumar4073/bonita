(function () {
  'use strict';
 
  function capitalize(text) {
    return text.charAt(0).toUpperCase() + text.slice(1);
  }
 
  angular.module('bonitasoft.ui.services').service('bindingService', function ($injector) {
    return {
      create: function (property, context) {
        var Binding = $injector.get(capitalize(property.type) + 'Binding');
        return new Binding(property, context);
      }
    };
  });
})();
